279     private void logParameters() {
280         if (getLog().isDebugEnabled()) {
281             getLog().debug("Job parameters:");
282             getLog().debug("\tskip = " + skip);
283             getLog().debug("\tverbose = " + verbose + createCalculatedInfo(Boolean.TRUE));
284             getLog().debug("\tforce = " + force);
285             getLog().debug("\tsourceDirectory = " + sourceDirectory);
286             getLog().debug("\toutputDirectory = " + outputDirectory);
287             getLog().debug("\tfilesetPatternFormat = " + filesetPatternFormat);
288             String calculated = includes.length != 0 ? "" : createCalculatedInfo(Arrays.toString(getDefaultIncludes()));
289             getLog().debug("\tincludes = " + Arrays.toString(includes) + calculated);
290             getLog().debug("\texcludes = " + Arrays.toString(excludes));
291             calculated = maxRules > 0 ? "" : createCalculatedInfo(Integer.MAX_VALUE);
292             getLog().debug("\tmaxRules = " + maxRules + calculated);
293             calculated = rulesLimit > 0 ? "" : createCalculatedInfo(Integer.MAX_VALUE);
294             getLog().debug("\trulesLimit = " + rulesLimit + calculated);
295             getLog().debug("\tstandard = " + standard);
296             getLog().debug("\tcacheTokenType = " + cacheTokenType);
297             getLog().debug("\tcacheTokenParameter = " + cacheTokenParameter);
298             calculated = "";
299             if (cacheTokenValue == null) {
300                 final String defaultCacheTokenValue = getDefaultCacheTokenValue();
301                 calculated = defaultCacheTokenValue != null ? createCalculatedInfo(defaultCacheTokenValue) : "";
302             }
303             getLog().debug("\tcacheTokenValue = " + cacheTokenValue + calculated);
304             getLog().debug("\tencoding = " + encoding);
305             getLog().debug("\toutputFileNamePattern = " + outputFileNamePattern);
306             getLog().debug("\toutputPartNamePattern = " + outputPartNamePattern);
307             getLog().debug("");
308         }
309     }

311     private static String createCalculatedInfo(final Object value) {
312         return String.format(" (calculated: %s)", value);
313     }

315     private String[] getDefaultIncludes() {
316         if (ScannerPatternFormat.ANT.name().equalsIgnoreCase(filesetPatternFormat)) {
317             return new String[] { "**/*.css" };
318         } else {
319             return new String[] { "^.+\\.css$" };
320         }
321     }
322 
323     private String getDefaultCacheTokenValue() {
324         if (TokenType.DATE.name().equalsIgnoreCase(cacheTokenType)) {
325             return "yyyyMMddHHmmss";
326         } else {
327             return null;
328         }
329     }

331     private void calculateParameters() {
332         if (getLog().isDebugEnabled()) {
333             verbose = true;
334         }
335         if (includes.length == 0) {
336             includes = getDefaultIncludes();
337         }
338         if (maxRules < 1) {
339             maxRules = Integer.MAX_VALUE;
340         }
341         if (rulesLimit < 1) {
342             rulesLimit = Integer.MAX_VALUE;
343         }
344         if (cacheTokenValue == null) {
345             cacheTokenValue = getDefaultCacheTokenValue();
346         }
347     }

349     private void validateParameters() throws MojoExecutionException {
350         if (cacheTokenValue == null && TokenType.CUSTOM.name().equalsIgnoreCase(cacheTokenType)) {
351             throw new MojoExecutionException("Parameter cacheTokenValue is required when cacheTokenType is equal to \"custom\"!");
352         }
353     }

355     public void execute() throws MojoExecutionException, MojoFailureException {
356         logParameters();
357         if (skip) {
358             getLog().info("Skipping job execution");
359             return;
360         }
361         calculateParameters();
362         validateParameters();
363         runSplitter();
364     }
365 
366     private void runSplitter() throws MojoFailureException {
367         if (!sourceDirectory.exists()) {
368             getLog().warn("Source directory does not exist: " + sourceDirectory.getAbsolutePath());
369             return;
370         }
371         final Collection<File> files = getFiles();
372         if (files.isEmpty()) {
373             getLog().warn("No sources to compile");
374             return;
375         }
376         splitFiles(files);
377     }
378 
379     private Collection<File> getFiles() {
380         final ScannerPatternFormat patternFormat = ScannerPatternFormat.toPattern(filesetPatternFormat);
381         final FileScanner scanner = new ScannerFactory().create(patternFormat, getLog());
382         if (verbose) {
383             getLog().info("Scanning directory for sources...");
384         }
385         return scanner.getFiles(sourceDirectory, includes, excludes);
386     }
387 
388     private void splitFiles(final Collection<File> sources) throws MojoFailureException {
389         final String sourceFilesText = "source file" + (sources.size() != 1 ? "s" : "");
390         getLog().info("Splitting " + sources.size() + ' ' + sourceFilesText + " to " + outputDirectory.getAbsolutePath());
391         final Timer timer = SystemTimer.getStartedTimer();
392         for (final File source : sources) {
393             if (isCompilationRequired(source)) {
394                 splitFile(source);
395             } else if (verbose) {
396                 getLog().info("Skipping CSS stylesheet split (not modified): " + source.getAbsolutePath());
397             }
398         }
399         getLog().info("Finished " + sourceFilesText + " split in " + timer.stop());
400     }
401 
402     private boolean isCompilationRequired(final File source) {
403         if (force) {
404             return true;
405         }
406         final File destination = new DestinationFileCreator(sourceDirectory, outputDirectory, outputFileNamePattern).create(source);
407         if (!destination.exists()) {
408             return true;
409         }
410         return source.lastModified() > destination.lastModified();
411     }
412 
413     private void splitFile(final File source) throws MojoFailureException {
414         Timer timer = null;
415         if (verbose) {
416             getLog().info("Splitting CSS stylesheet: " + source.getAbsolutePath());
417             timer = SystemTimer.getStartedTimer();
418         }
419         final String css = readCss(source);
420         final List<StyleSheet> parts;
421         try {
422             final StyleSheet stylesheet = new SteadyStateParser(getLog()).parse(css, Standard.create(standard));
423             new RulesLimitValidator(rulesLimit).validate(stylesheet);
424             new StylePropertiesLimitValidator(maxRules).validate(stylesheet);
425             parts = new Splliter(maxRules).split(stylesheet);
426         } catch (final Exception e) {
427             throw new MojoFailureException(e.getMessage(), e);
428         }
429         saveParts(source, parts);
430         if (timer != null) {
431             getLog().info("Finished in " + timer.stop());
432         }
433     }

435     private String readCss(final File file) throws MojoFailureException {
436         try {
437             return FileUtils.readFileToString(file, encoding);
438         } catch (final IOException e) {
439             throw new MojoFailureException("Cannot read file: " + file.getAbsolutePath(), e);
440         }
441     }
442 
443     private void saveParts(final File source, final List<StyleSheet> parts) throws MojoFailureException {
444         final DestinationFileCreator fileCreator = new DestinationFileCreator(sourceDirectory, outputDirectory);
445         fileCreator.setFileNamePattern(outputFileNamePattern);
446         final File file = fileCreator.create(source);
447         if (parts.size() == 1) {
448             saveCss(file, parts.get(0).toString());
449             return;
450         }
451 
452         // TODO add maxImports per file
453         final StringBuilder imports = new StringBuilder();
454         final String cacheToken = createCacheToken();
455         for (int index = 0; index < parts.size(); ++index) {
456             final String filePartFormat = outputPartNamePattern.replace(PART_INDEX_PARAMETER, String.valueOf(index + 1));
457             fileCreator.setFileNamePattern(filePartFormat);
458             final File filePart = fileCreator.create(source);
459             imports.append("@import \"");
460             imports.append(filePart.getName());
461             imports.append(cacheToken);
462             imports.append("\";\n");
463             saveCss(filePart, parts.get(index).toString());
464         }
465         saveCss(file, imports.toString());
466     }
467 
468     private String createCacheToken() {
469         if (TokenType.NONE.name().equalsIgnoreCase(cacheTokenType)) {
470             return "";
471         } else {
472             final String value = TokenType.create(cacheTokenType).createFactory().create(cacheTokenValue);
473             final StringBuilder cacheToken = new StringBuilder();
474             cacheToken.append('?');
475             cacheToken.append(UrlEscaper.escape(cacheTokenParameter));
476             cacheToken.append('=');
477             cacheToken.append(UrlEscaper.escape(value));
478             return cacheToken.toString();
479         }
480     }
481 
482     private void saveCss(final File file, final String css) throws MojoFailureException {
483         if (verbose) {
484             getLog().info("Saving CSS code to " + file.getAbsolutePath());
485         }
486         try {
487             FileUtils.write(file, css, encoding);
488         } catch (final IOException e) {
489             throw new MojoFailureException("Cannot save file: " + file.getAbsolutePath(), e);
490         }
491     }
